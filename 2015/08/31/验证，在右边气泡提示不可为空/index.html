验证，在右边气泡提示不可为空

发表于 2015-08-31   |   分类于 Jquery

/* ****************************************
*
* 支持text selectcheckbox readio，(value为空才提示必埴) 
* 具体用法:
* <input class="validatebox" validate="{required: true, title: '请输入邮箱地址', validtype:'email',validtitle:'邮箱格式不正确'}  />
* validtype的类型有：number，float，string，email， phone，url，date；
* minlength:最小长度,maxlength:最大长度,length:长度限制
具体请查看函数：validateInput(validType,value)
若需要增加其它类型验证，可以此地方添加

* 提示语会在td后面显示（以气泡提示）
* 以class "validatebox"为标志扫描
* required: true"为必填(若为不必填，但要验证格式，则只需要写验证格式的配置
如： validate="{validtype:email,validtitle:'邮箱格式不正确'}"
* 当为空时显示的提示语：title：'请输入奖品链接'
* 验证类型：validtype:'url'
* 验证不能过时显示的提示语：validtitle: '链接地址格式不正确'
*
*  if(checkbox or radio) 页面标签格式如下：
*  <div style="float:left;" class="validatebox" inputType="checkbox" required="true" title="checkbox">
*     <input type="checkbox" value=""/>是
*     <input type="checkbox" value=""/>否 
*  </div>
*
* 引用此js，且在相应页面的js里调用如下：
*       
*     var IsValid = true;       
*     //验证表单输入值
*     IsValid = validateBoxInput();
*验证radio,check
* ****************************************/


/*验证函数
*@param container 验证元素所在的form，用jquery选择器写法,例子：".xxForm"，"#xx form"
*@param isTip 是否通过Tip弹出框提示，默认为是。为否时，不弹出信息，输入框背景颜色改变同时title属性变成提示语
*@return 返回整个表单是否通过了验证
*/
function validateBoxInput(container, isTip) {
$validateItems = container != undefined ? $(container + " .validatebox") : $(".validatebox");
isTip = (isTip == undefined) ? true : isTip;

var IsValid = true;
$validateItems.each(function () {
    $this = $(this);
    var isItemValid = validateItem($this, isTip);
    if (!isItemValid) {
        IsValid = false;
    }
    $this.blur(function () {
        validateItem($(this), isTip);
    });
});
return IsValid;
}

function validateItem($this, isTip) {
var validate = $this.attr("validate");
validate = eval("(" + validate + ")");  // 字符串转json
var IsValid = true;
$this.next(".vMsg").remove();
if (validate.required) {
    var errTitle = validate.title != undefined ? validate.title : '不能为空';
    //验证radio,checkBox
    if (validate.itype == 'radio' || validate.itype == 'checkbox') {
        if (!validateCheckRadio(validate.itype, errTitle)) {
            IsValid = false;
        }
    }
    else if ($.trim($this.val()) == "") { // 当值为空的时候才提示 
        if (isTip) {
            $this.after("<span class='vMsg'>" + errTitle + "</span>");
        } else {
            $this.addClass("l-text-invalid").attr("title", errTitle);
        }
        IsValid = false;
    }
}
// 当不要求必填时，验证输入的值格式是否正确
if (IsValid && validate.validtype != undefined && $.trim($this.val()) != "" && !validateInput(validate.validtype, $.trim($this.val()))) {

    var validateTitle = validate.validtitle != undefined ? validate.validtitle : '格式不正确';
    if (isTip) {
        $this.after("<span class='vMsg'>" + validateTitle + "</span>");
    } else {
        $this.addClass("l-text-invalid").attr("title", validateTitle);
    }
    IsValid = false;
}
//验证长度
if (IsValid && validate.length != undefined && $.trim(validate.length) != "") {
    var len = parseInt(validate.length);
    if (len > 0) {
        if ($.trim($this.val()).length != len) {
            if (isTip) {
                //$this.ligerTip({ content: "长度为" + len + "位", width: 100 });
                $this.after("<span class='vMsg'>" + "长度为" + len + "位" + "</span>");
            } else{
                $this.addClass("l-text-invalid").attr("title", "长度为" + len + "位");
            }
            IsValid = false;
        }
    }
}
//验证最小最大长度
if (IsValid && (validate.minlength != undefined || validate.maxlength != undefined)) {

    var minlength = validate.minlength!=undefined?parseInt(validate.minlength):-1;
    var maxlength = validate.maxlength != undefined ? parseInt(validate.maxlength) : -1;
    var valLen = $.trim($this.val()).length;
    if (valLen > 0) {
        var errMsg = "";
        if (minlength > 0 && valLen < minlength) {
            errMsg = maxlength > 0 ? "请输入" + minlength + "到" + maxlength + "个字符" : "最小长度为" + minlength;
        } else if (maxlength > 0 && valLen > maxlength) {
            errMsg = minlength > 0 ? "请输入" + minlength + "到" + maxlength + "个字符" : "最大长度为" + maxlength;
        }
        if (errMsg != "") {
            if (isTip) {
                $this.after("<span class='vMsg'>" + errMsg + "</span>");
            } else{
                $this.addClass("l-text-invalid").attr("title", errMsg);
            }
            IsValid = false;
        } 
    }
}

if (IsValid) {
    if (isTip) {
        $this.next(".vMsg").remove();
    } else{
        $this.removeClass("l-text-invalid").attr("title", "");
    }
}
return IsValid;
}

//验证radio,checkBox
function validateCheckRadio(inputType, title) {
if ($this.find(":" + inputType + ":checked").length <= 0) {
    $this.after("<span class='vMsg'>" + title + "</span>");
    return false;
}
else {
    $this.next(".vMsg").remove();
    return true;
}
}

// 验证输入格式是否正确
function validateInput(validType, value) {

switch (validType) {
    case "uinteger": //正整数
        return checkUnsignedInteger(value);
        break;
    case "number": //数字
        return checkNumber(value);
        break;
    case "float":
        return checkFloat(value);
        break;
    case "string":
        return checkEnglistNumber(value);
        break;
    case "email":
        return checkEmail(value);
        break;
    case "phone":
        return checkPhone(value);
        break;
    case "url":
        return checkUrlHttp(value);
        break;
    case "date":
        return checkDate(value);
        break;
    case "select":
        return value != '-1';
        break;
}
}

// 网站地址url验证
// ====================
function checkUrl(strValue) {
    var reg = new RegExp("^http(s)?://([\w-]+\.)+[\w-]+(/.*)?");

    if (!reg.test(strValue))
        return false;

    return true;
}

//
// 完美的js验证网址url（正则表达式）
//
function IsUrl(strValue) {
    //var strRegex = "^((https|http|ftp|rtsp|mms)?://)"
    var strRegex = "^http(s)?://"
    + "(([0-9a-z_!~*'().&=+$%-]+: )?[0-9a-z_!~*'().&=+$%-]+@)?" //ftp的user@  
    + "(([0-9]{1,3}\.){3}[0-9]{1,3}"                             // IP形式的URL- 199.194.52.184  
    + "|"                                                        // 允许IP和DOMAIN（域名） 
    + "([0-9a-z_!~*'()-]+\.)*"                                   // 域名- www.  
    + "([0-9a-z][0-9a-z-]{0,61})?[0-9a-z]\."                     // 二级域名  
    + "[a-z]{2,6})"                                              // first level domain- .com or .museum  
    + "(:[0-9]{1,4})?"                                           // 端口- :80  
    + "((/?)|"                                                   // a slash isn't required if there is no file name  
    + "(/[0-9a-z_!~*'().;?:@&=+$,%#-]+)+/?)$";
    var re = new RegExp(strRegex);
    if (re.test(strValue)) {
        return (true);
    } else {
        return (false);
    }
}


// ====================
// 日期验证
// ====================
function checkDate(strValue) {
    //var reg /^\d{4}(\-|\/|\.)\d{1,2}\1\d{1,2}$/
    var reg = /^(\d{4})-(((0?)[1-9])|11|12)-(((0?)[1-9])|([1-3][0-9]))$/;
    return reg.test(strValue);
}
